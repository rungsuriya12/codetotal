Read

รวมคำสั่งการ Read DATA และเงือนไขการเรียกใช้งานเบื้องต้น.

SELECT * FROM employees

การค้นหาข้อมูลทั้งหมดที่อยู่ในตารางเครื่องหมาย * คือการแสดงข้อมูลทั้งหมดที่อยู่ในตาราง employees

SELECT id,first_name,last_name FROM employees

สามารถเลือกให้ แสดงเฉพาะ Column ที่ต้องการได้โดยการระบุ ชื่อ Column ที่ต้องการจะแสดงข้อมูล เฉพาะ Column ที่เราได้กำหนดไว้ในตาราง

SELECT * FROM employees WHERE job_position = 'Graphic Designer'

ใช้เงือนไขในการค้นหาข้อมูลจากตาราง employees ที่มี job_position เท่ากับ Graphic Designer จะแสดงข้อมูลพนักงานที่มี ตำแหน่ง ตรงกับ Graphic Designer เท่านั้น

SELECT * FROM employees WHERE job_position = 'Graphic Designer' AND gender = 'Male'

คำสั่ง AND และ OR ใช้ร่วมกับคำสั่ง WHERE เพื่อกำหนดเงื่อนไข มากกว่า 1 เงื่อนไข เช่น ต้องการ ค้นหาข้อมูลจากตาราง employees แสดงข้อมูลพนักงานที่มีตำแหน่งตรงกับ Graphic Designer และ เป็นผู้ชายเท่านั้น

SELECT * FROM employees WHERE job_position = 'Financial Advisor' OR job_position = 'Marketing Manager'

แสดงข้อมูล พนักงานที่มีตำแหน่ง ตรงกับ Financial Advisor หรือ Marketing Manager

SELECT * FROM `employees` WHERE salary BETWEEN 20000 AND 25000

คำสั่ง between คือคำสั่งสำหรับการค้นหาข้อมูลที่มีค่าระหว่างค่าเริ่มต้น และ สิ้นสุด แสดงข้อมูล พนักงานที่มีเงินเดือนระหว่าง 20000 ถึง 25000

SELECT * FROM employees WHERE salary BETWEEN 20000 AND 25000 AND job_position = 'Help Desk Operator' AND gender = 'Female'

เพิ่มเงือนไขในการหาพนักงาน ที่มีเงินเดือนระหว่าง 20000 ถึง 25000 และ มีตำแหน่ง Help Desk Operator และ เป็นผู้หญิง

SELECT DISTINCT (job_position) FROM employees

DISTINCT คือ คำสั่งสำหรับ แสดงข้อมูล ที่ไม่ซ้ำกัน โดย ถ้าหากมีข้อมูลที่ซ้ำกัน จะแสดง เพียง Records เดียว จะแสดงข้อมูลใน job_position ทั้งหมดที่มีข้อมูลไม่ซ้ำกัน

SELECT * FROM employees ORDER BY first_name ASC

ORDER BY คือคำสั่งสำหรับการ เรียงลำดับข้อมูล Column ที่เลือกโดยสามารถกำหนดการเรียงลำดับได้ 2 รูปแบบ คือ แบบ desc (descending ) คือการเรียงลำดับจากมากไปน้อย แบบ asc (ascending ) คือ การเรียบลำดับจากน้อยไปมาก แต่หากไม่กำหนด รูปแบบ ค่าเริ่มต้น จะเรียง แบบ asc ตัวอย่าง ต้องการแสดงข้อมูล ชื่อพนักงาน (first_name) โดยเรียบลำดับจาก น้อยไปหามาก

SELECT * FROM employees WHERE first_name LIKE "%an%"

LIKE คือคำสั่งค้นหาข้อความหรือ ตัวเลขภายใน Column ที่มีข้อมูลภายในตรงกับที่กำหนด รูปแบบคำสั่งใช้เครื่องหมาย % (percent) เพื่อกำหนดรูปแบบในการค้นหาข้อมูล ตัวอย่าง แสดงข้อมูลทีมีตัวอักษร an เป็นส่วนประกอบอยู่ด้วย

SELECT * FROM `employees` WHERE first_name LIKE "D%"

ตัวอย่าง แสดงข้อมูล ที่มี ตัวอักษร D อยู่เป็นลำดับแรก แสดงข้อมูล ใน column first_name ทีมีคำว่า D อยู่เป็นลำดับแรก

SELECT * FROM employees WHERE first_name LIKE "%E"

ตัวอย่าง แสดงข้อมูล ที่มีตัวอักษร E อยู่ลำดับสุดท้าย ผลการทำงาน แสดงข้อมูล ใน column first_name ทีมีคำว่า E อยู่เป็นลำดับสุดท้าย

SELECT * FROM employees WHERE company IN('Skinder','Fiveclub')

IN คือคำสั่งในการค้นหาข้อมูลภายใน Column ที่กำหนด ตัวอย่าง ต้องการค้นหาข้อมูลในพนักงานที่ มีข้อมูล company คือ Skinder หรือ Fiveclub

SELECT COUNT(*) FROM employees

COUNT() คือ Function สำหรับนับจำนวนซึ่งจะ แสดงค่าจำนวนทั้งหมดของข้อมูล Column ที่กำหนด ตัวอย่าง ต้องการนับจำนวน พนักงานทั้งหมดในตาราง employees

SELECT COUNT(*) FROM employees WHERE job_position = 'Marketing Manager'

ผลการทำงาน แสดงจำนวนพนักงานที่มีตำแหน่ง คือ Marketing Manager

SELECT job_position,COUNT(*) FROM employees GROUP BY job_position

GROUP BY คือคำสั่งสำหรับจัดกลุ่มข้อมูลโดยสามารถใช้คำสั่ง SQL Function ทาง คณิตศาสตร์ เพื่อคำนวณข้อมูลที่ถูกจัดกลุ่มไว้ได้ ตัวอย่าง ต้องการแสดงข้อมูลตำแหน่งและ จำนวนพนักงานในแต่ละตำแหน่ง
